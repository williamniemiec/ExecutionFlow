package examples.overloadedMethod;                                             
public class OverloadClass{                                                    
     private int num;                                                          
     @wniemiec.executionflow.runtime.CollectCalls public OverloadClass(int num){        
          this.num = num;                                                      
     }                                                                         
     @wniemiec.executionflow.runtime.CollectCalls public OverloadClass(){               
          this(0);                                                             
     }                                                                         
     @wniemiec.executionflow.runtime.CollectCalls public int overloadedMethod(int num){ 
          num *= 10;                                                           
          return num;                                                          
     }                                                                         
     @wniemiec.executionflow.runtime.CollectCalls public int overloadedMethod(){        
          int x = 0;                                                           
          x++;                                                                 
          x++;                                                                 
          return x+overloadedMethod(10);                                       
     }                                                                         
}
