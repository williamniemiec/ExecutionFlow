package examples.innerClass;                                                  
public class OuterClass{                                                      
     private String text ;                                                    
     @executionflow.runtime.CollectCalls public OuterClass(String text){      
          this.text = text;                                                   
     }                                                                        
     public class InnerClass{                                                 
          private String text;                                                
          @executionflow.runtime.CollectCalls public InnerClass(String text){ 
               this.text = text;                                              
          }                                                                   
          @executionflow.runtime.CollectCalls public String getText(){        
               return this.text;                                              
          }                                                                   
     }                                                                        
     @executionflow.runtime.CollectCalls public String getText(){             
          return this.text;                                                   
     }                                                                        
}
